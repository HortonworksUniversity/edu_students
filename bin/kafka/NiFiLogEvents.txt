### Open Terminal A
kinit kafka/EDU

# Set path to Kafka bin
export PATH="$PATH:/usr/hdp/current/kafka-broker/bin"

## NiFiLogEvents
# Deleting a topic
kafak-topics.sh --delete --zookeeper master01.cloudair.lan:2181 --topic NiFiLogEvents

# Create topics
kafka-topics.sh --create --zookeeper master01.cloudair.lan:2181 --replication-factor 3 --partitions 2 --topic NiFiLogEvents

# List topics
kafka-topics.sh --list --zookeeper master01.cloudair.lan:2181

# Describe topic
kafka-topics.sh --zookeeper master01.cloudair.lan:2181 --describe --topic NiFiLogEvents 

# Unsecured view messages
kafka-console-consumer.sh --bootstrap-server flow01.cloudair.lan:6667 --topic NiFiLogEvents --from-beginning

# Secured view messages
kafka-console-consumer.sh --bootstrap-server flow01.cloudair.lan:6668 --topic NiFiLogEvents --from-beginning --consumer.config /var/local/conf/client-ssl.properties
